(define_insn "cv_abs"
    [(set (match_operand:SI 0 "register_operand" "=r")
      (if_then_else:SI
        (lt:SI (match_operand:SI 1 "register_operand" "r")
               (const_int 0))
        (neg:SI (match_dup 1))
        (match_dup 1)
      )
  )]
  "TARGET_COREV_ALU"
  "cv.abs\t%0,%1"
  [(set_attr "type" "imul")
   (set_attr "mode" "SI")]
)

(define_insn "cv_clip"
  [(if_then_else:SI                                              ; if
    (le:SI (match_operand 1 "register_operand" "r")              ; rs1 <= -2^(Is2-1)
           (neg:SI (ashift:SI
                     (const_int 1)
                     (minus:SI (match_operand 2 "immediate_operand" "i")
                               (const_int 1)
                     )
                  )
           )
    )
    (set (match_operand:SI 0 "register_operand" "=r")
         (neg:SI (ashift:SI
                   (const_int 1)
                   (minus:SI (match_dup 2)
                             (const_int 1)
                   )
                )
         )
    )                                                            ; then rD = -2^(Is2-1)
    (if_then_else:SI                                             ; else, if
      (ge:SI (match_dup 1)                                       ; rs1 >= 2^(ls2â€“1)-1
             (minus:SI (ashift:SI
                         (const_int 1)
                         (minus:SI (match_dup 2)
                                   (const_int 1)
                         )
                      )
                      (const_int 1)
            )
      )
      (set (match_dup 0)                                         ; then rd = 2^(Is2-1)-1
        (minus:SI (ashift:SI
                    (const_int 1)
                    (minus:SI (match_dup 2)
                              (const_int 1)
                    )
                  )
                  (const_int 1)
        )
      )
      (set (match_dup 0) (match_dup 1))                         ; else rd = rs1
    )
   )]
   "TARGET_COREV_ALU"
   "cv.clip\t%0,%1,%2"
   [(set_attr "type" "arith")
    (set_attr "mode" "SI")]
 )

(define_insn "cv_clipr"
  [(if_then_else:SI                                              ; if
    (le:SI (match_operand 1 "register_operand" "r")              ; rs1 <= -2^(Is2-1)
           (neg:SI (ashift:SI
                     (const_int 1)
                     (minus:SI (match_operand 2 "immediate_operand" "i")
                               (const_int 1)
                     )
                   )
           )
    )
    (set (match_operand:SI 0 "register_operand" "=r")
         (neg:SI (ashift:SI
                   (const_int 1)
                   (minus:SI (match_dup 2)
                             (const_int 1)
                   )
                 )
         )
    )                                                   
    (if_then_else:SI                                    
      (ge:SI (match_dup 1)                               
             (match_dup 2)
      )
      (set (match_dup 0)                                  
           (match_dup 2)
      )
      (set (match_dup 0) (match_dup 1))                    
    )
   )]
   "TARGET_COREV_ALU"
   "cv.clipr\t%0,%1,%2"
   [(set_attr "type" "arith")
    (set_attr "mode" "SI")]
 )
